18  package com.google.code.play2.plugin;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.util.Arrays;
23  
24  import org.apache.maven.plugin.MojoExecutionException;
25  import org.apache.maven.plugins.annotations.Component;
26  import org.apache.maven.plugins.annotations.LifecyclePhase;
27  import org.apache.maven.plugins.annotations.Mojo;
28  import org.apache.maven.plugins.annotations.Parameter;
29  
30  import org.sonatype.plexus.build.incremental.BuildContext;
31  
32  import com.google.code.play2.provider.api.AssetCompilationException;
33  import com.google.code.play2.provider.api.CoffeescriptCompilationResult;
34  import com.google.code.play2.provider.api.Play2CoffeescriptCompiler;
35  import com.google.code.play2.provider.api.Play2JavascriptCompiler;
36  import com.google.code.play2.provider.api.Play2Provider;
37  
38  /**
39   * Compile Coffee Script assets
40   * 
41   * @author <a href="mailto:gslowikowski@gmail.com">Grzegorz Slowikowski</a>
42   * @since 1.0.0
43   */
44  @Mojo( name = "coffee-compile", defaultPhase = LifecyclePhase.GENERATE_RESOURCES )
45  public class Play2CoffeeCompileMojo
46      extends AbstractPlay2AssetsCompileMojo
47  {
48      /**
49       * CoffeeScript compiler entry points includes, separated by commas.
50       * 
51       * @since 1.0.0
52       */
53      @Parameter( property = "play2.coffeeEntryPointsIncludes", defaultValue = "**/*.coffee" )
54      private String coffeeEntryPointsIncludes;
55  
56      /**
57       * CoffeeScript compiler entry points excludes, separated by commas.
58       * 
59       * @since 1.0.0
60       */
61      @Parameter( property = "play2.coffeeEntryPointsExcludes", defaultValue = "" )
62      private String coffeeEntryPointsExcludes;
63  
64      /**
65       * CoffeeScript compiler options, separated by spaces.
66       * 
67       * @since 1.0.0
68       */
69      @Parameter( property = "play2.coffeescriptOptions", defaultValue = "" )
70      private String coffeescriptOptions;
71  
72      /**
73       * For M2E integration.
74       */
75      @Component
76      private BuildContext buildContext;
77  
78      @Override
79      protected String getAssetsIncludes()
80      {
81          return coffeeEntryPointsIncludes;
82      }
83  
84      @Override
85      protected String getAssetsExcludes()
86      {
87          return coffeeEntryPointsExcludes;
88      }
89  
90      @Override
91      protected void compileAssets( File assetsSourceDirectory, String[] fileNames, File outputDirectory )
92          throws AssetCompilationException, IOException, MojoExecutionException
93      {
94          int compiledFiles = 0;
95  
96          Play2Provider play2Provider = getProvider();
97          Play2CoffeescriptCompiler compiler = play2Provider.getCoffeescriptCompiler();
98          if ( coffeescriptOptions != null )
99          {
100             compiler.setCompilerOptions( Arrays.asList( coffeescriptOptions.split( " " ) ) );
101         }
102 
103         for ( String fileName : fileNames )
104         {
105             File coffeeFile = new File( assetsSourceDirectory, fileName );
106 
107             String jsFileName = fileName.replace( ".coffee", ".js" );
108             File jsFile = new File( outputDirectory, jsFileName );
109 
110             String minifiedJsFileName = fileName.replace( ".coffee", ".min.js" );
111             File minifiedJsFile = new File( outputDirectory, minifiedJsFileName );
112 
113             boolean modified = true;
114             if ( jsFile.isFile() && minifiedJsFile.isFile() )
115             {
116                 modified =
117                     ( jsFile.lastModified() < coffeeFile.lastModified() && minifiedJsFile.lastModified() < coffeeFile.lastModified() );
118             }
119 
120             if ( modified )
121             {
122                 getLog().debug( String.format( "Processing \"%s\"", fileName ) );
123 
124                 CoffeescriptCompilationResult result = compiler.compile( coffeeFile );
125                 String jsContent = result.getJs();
126                 createDirectory( jsFile.getParentFile(), false );
127                 writeToFile( jsFile, jsContent );
128                 buildContext.refresh( jsFile );
129                 try
130                 {
131                     Play2JavascriptCompiler jsCompiler = play2Provider.getJavascriptCompiler();
132                     String minifiedJsContent = jsCompiler.minify( jsContent, coffeeFile.getName() );
133                     // String minifiedJsContent = JavascriptCompiler.minify( jsContent, coffeeFile.getName() );
134                     createDirectory( minifiedJsFile.getParentFile(), false );
135                     writeToFile( minifiedJsFile, minifiedJsContent );
136                     compiledFiles++;
137                     buildContext.refresh( minifiedJsFile );
138                 }
139                 catch ( AssetCompilationException e )
140                 {
141                     if ( minifiedJsFile.exists() && minifiedJsFile.isFile() && minifiedJsFile.delete() )
142                     {
143                         buildContext.refresh( minifiedJsFile );
144                     }
145                 }
146             }
147             else
148             {
149                 getLog().debug( String.format( "\"%s\" skipped - no changes", fileName ) );
150             }
151         }
152 
153         getLog().info( String.format( "%d assets processed, %d compiled", Integer.valueOf( fileNames.length ),
154                                       Integer.valueOf( compiledFiles ) ) );
155     }
156 
157 }
