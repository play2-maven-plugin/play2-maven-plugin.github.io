135     @Override
136     public void execute()
137         throws MojoExecutionException, MojoFailureException
138     {
139         if ( !"pom".equals( project.getPackaging() ) )
140         {
141             try
142             {
143                 long ts = System.currentTimeMillis();
144                 internalExecute();
145                 long te = System.currentTimeMillis();
146                 getLog().debug( String.format( "Mojo execution time: %d ms", Long.valueOf( te - ts ) ) );
147             }
148             catch ( IOException e )
149             {
150                 throw new MojoExecutionException( "?", e );
151             }
152         }
153     }

158     protected final BufferedReader createBufferedFileReader( File file, String encoding )
159         throws FileNotFoundException, UnsupportedEncodingException
160     {
161         return new BufferedReader( new InputStreamReader( new FileInputStream( file ), encoding ) );
162     }
163 
164     protected final BufferedWriter createBufferedFileWriter( File file, String encoding )
165         throws FileNotFoundException, UnsupportedEncodingException
166     {
167         return new BufferedWriter( new OutputStreamWriter( new FileOutputStream( file ), encoding ) );
168     }
169 
170     protected String readFileFirstLine( File file )
171         throws IOException
172     {
173         String result = null;
174 
175         BufferedReader is = createBufferedFileReader( file, "UTF-8" );
176         try
177         {
178             result = is.readLine();
179         }
180         finally
181         {
182             is.close();
183         }
184         return result;
185     }
186 
187     protected void writeToFile( File file, String line )
188         throws IOException
189     {
190         BufferedWriter writer = createBufferedFileWriter( file, "UTF-8" );
191         try
192         {
193             writer.write( line );
194         }
195         finally
196         {
197             writer.flush();
198             writer.close();
199         }
200     }

202     protected void createDirectory( File directory, boolean overwrite )
203         throws IOException
204     {
205         if ( directory.exists() )
206         {
207             if ( directory.isDirectory() )
208             {
209                 if ( overwrite )
210                 {
211                     FileUtils.cleanDirectory( directory );
212                 }
213             }
214             else
215             // file if ( directory.isFile() )
216             {
217                 getLog().info( String.format( "Deleting \"%s\" file", directory ) ); // TODO-more descriptive message
218                 if ( !directory.delete() )
219                 {
220                     throw new IOException( String.format( "Cannot delete \"%s\" file", directory.getCanonicalPath() ) );
221                 }
222             }
223         }
224 
225         if ( !directory.exists() )
226         {
227             if ( !directory.mkdirs() )
228             {
229                 throw new IOException( String.format( "Cannot create \"%s\" directory", directory.getCanonicalPath() ) );
230             }
231         }
232     }

234     // Cached classloaders
235     private static final ConcurrentHashMap<String, ClassLoader> cachedClassLoaders = new ConcurrentHashMap<String, ClassLoader>( 2 );
236 
237     private static ClassLoader getCachedClassLoader( String providerId )
238     {
239         return cachedClassLoaders.get( providerId );
240     }
241 
242     private static void setCachedClassLoader( String providerId, ClassLoader classLoader )
243     {
244         cachedClassLoaders.put( providerId, classLoader );
245     }

247     protected Play2Provider getProvider()
248         throws MojoExecutionException
249     {
250         Play2Provider provider = null;
251         if ( !play2Providers.isEmpty() )
252         {
253             provider = getDeclaredProvider();
254         }
255         else
256         {
257             provider = getWellKnownProvider();
258         }
259         return provider;
260     }
261 
262     private Play2Provider getDeclaredProvider()
263         throws MojoExecutionException
264     {
265         if ( play2Providers.size() > 1 )
266         {
267             throw new MojoExecutionException( "Too many Play! providers defined. A maximum of one allowed." );
268         }
269 
270         Map.Entry<String, Play2Provider> providerEntry = play2Providers.entrySet().iterator().next();
271         String providerId = providerEntry.getKey();
272         Play2Provider provider = providerEntry.getValue();
273 
274         getLog().debug( String.format( "Using declared provider \"%s\".", providerId ) );
275 
276         return provider;
277     }

279     private Play2Provider getWellKnownProvider()
280         throws MojoExecutionException
281     {
282         try
283         {
284             String providerId = Play2Providers.getDefaultProviderId( playVersion );
285             ClassLoader providerClassLoader = getCachedClassLoader( providerId );
286             if ( providerClassLoader == null )
287             {
288                 getLog().debug( String.format( "Cached classloader for provider \"%s\" not available.", providerId ) );
289             }
290             else
291             {
292                 if ( providerClassLoader.getParent() == Thread.currentThread().getContextClassLoader() )
293                 {
294                     getLog().debug( String.format( "Using cached classloader for provider \"%s\".", providerId ) );
295                 }
296                 else
297                 {
298                     getLog().debug( String.format( "Invalidated cached classloader for provider \"%s\". Parent classloader changed from %d to %d.",
299                                                    providerId,
300                                                    Integer.valueOf( providerClassLoader.getParent().hashCode() ),
301                                                    Integer.valueOf( Thread.currentThread().getContextClassLoader().hashCode() ) ) );
302                     providerClassLoader = null;
303                 }
304             }
305 
306             if ( providerClassLoader == null )
307             {
308                 Artifact providerArtifact =
309                     getResolvedArtifact( pluginGroupId, "play2-provider-" + providerId, pluginVersion );
310 
311                 Set<Artifact> providerDependencies = getAllDependencies( providerArtifact );
312                 List<File> classPathFiles = new ArrayList<File>( providerDependencies.size() + 2 );
313                 classPathFiles.add( providerArtifact.getFile() );
314                 for ( Artifact dependencyArtifact : providerDependencies )
315                 {
316                     classPathFiles.add( dependencyArtifact.getFile() );
317                 }
318 
319                 List<URL> classPathUrls = new ArrayList<URL>( classPathFiles.size() );
320                 for ( File classPathFile : classPathFiles )
321                 {
322                     classPathUrls.add( new URL( classPathFile.toURI().toASCIIString() ) );
323                 }
324 
325                 providerClassLoader =
326                     new URLClassLoader( classPathUrls.toArray( new URL[classPathUrls.size()] ),
327                                         Thread.currentThread().getContextClassLoader() );
328                 getLog().debug( String.format( "Setting cached classloader for provider \"%s\" with parent classloader %d",
329                                                providerId, Integer.valueOf( providerClassLoader.getParent().hashCode() ) ) );
330                 setCachedClassLoader( providerId, providerClassLoader );
331             }
332 
333             String providerClassName =
334                 String.format( "com.google.code.play2.provider.%s.%sProvider", providerId,
335                                providerId.substring( 0, 1 ).toUpperCase( Locale.ROOT ) + providerId.substring( 1 ) );
336             Play2Provider provider = (Play2Provider) providerClassLoader.loadClass( providerClassName ).newInstance();
337 
338             getLog().debug( String.format( "Using autodetected provider \"%s\".", providerId ) );
339 
340             return provider;
341         }
342         catch ( ArtifactNotFoundException e )
343         {
344             throw new MojoExecutionException( "Provider autodetection failed", e );
345         }
346         catch ( ArtifactResolutionException e )
347         {
348             throw new MojoExecutionException( "Provider autodetection failed", e );
349         }
350         catch ( ClassNotFoundException e )
351         {
352             throw new MojoExecutionException( "Provider autodetection failed", e );
353         }
354         catch ( IllegalAccessException e )
355         {
356             throw new MojoExecutionException( "Provider autodetection failed", e );
357         }
358         catch ( InstantiationException e )
359         {
360             throw new MojoExecutionException( "Provider autodetection failed", e );
361         }
362         catch ( InvalidDependencyVersionException e )
363         {
364             throw new MojoExecutionException( "Provider autodetection failed", e );
365         }
366         catch ( MalformedURLException e )
367         {
368             throw new MojoExecutionException( "Provider autodetection failed", e );
369         }
370         catch ( ProjectBuildingException e )
371         {
372             throw new MojoExecutionException( "Provider autodetection failed", e );
373         }
374     }

378     protected/*private*/ Artifact getResolvedArtifact( String groupId, String artifactId, String version )
379         throws ArtifactNotFoundException, ArtifactResolutionException
380     {
381         Artifact artifact = factory.createArtifact( groupId, artifactId, version, Artifact.SCOPE_RUNTIME, "jar" );
382         resolver.resolve( artifact, remoteRepos, localRepo );
383         return artifact;
384     }
385 
386     protected/*private*/ Set<Artifact> getAllDependencies( Artifact artifact )
387         throws ArtifactNotFoundException, ArtifactResolutionException, InvalidDependencyVersionException,
388         ProjectBuildingException
389     {
390         Set<Artifact> result = new HashSet<Artifact>();
391         MavenProject p = projectBuilder.buildFromRepository( artifact, remoteRepos, localRepo );
392         Set<Artifact> d = resolveDependencyArtifacts( p );
393         result.addAll( d );
394         for ( Artifact dependency : d )
395         {
396             Set<Artifact> transitive = getAllDependencies( dependency );
397             result.addAll( transitive );
398         }
399         return result;
400     }

402     /**
403      * This method resolves the dependency artifacts from the project.
404      * 
405      * @param theProject The POM.
406      * @return resolved set of dependency artifacts.
407      * @throws ArtifactResolutionException
408      * @throws ArtifactNotFoundException
409      * @throws InvalidDependencyVersionException
410      */
411     private Set<Artifact> resolveDependencyArtifacts( MavenProject theProject )
412         throws ArtifactNotFoundException, ArtifactResolutionException, InvalidDependencyVersionException
413     {
414         AndArtifactFilter filter = new AndArtifactFilter();
415         filter.add( new ScopeArtifactFilter( Artifact.SCOPE_TEST ) );
416         filter.add( new NonOptionalArtifactFilter() );
417         // TODO follow the dependenciesManagement and override rules
418         Set<Artifact> artifacts = theProject.createArtifacts( factory, Artifact.SCOPE_RUNTIME, filter );
419         for ( Artifact artifact : artifacts )
420         {
421             resolver.resolve( artifact, remoteRepos, localRepo );
422         }
423         return artifacts;
424     }

426     private static class NonOptionalArtifactFilter
427         implements ArtifactFilter
428     {
429         @Override
430         public boolean include( Artifact artifact )
431         {
432             return !artifact.isOptional();
433         }
