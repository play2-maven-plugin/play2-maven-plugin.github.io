132     @Override
133     public void execute()
134         throws MojoExecutionException, MojoFailureException
135     {
136         if ( !"pom".equals( project.getPackaging() ) )
137         {
138             try
139             {
140                 long ts = System.currentTimeMillis();
141                 internalExecute();
142                 long te = System.currentTimeMillis();
143                 getLog().debug( String.format( "Mojo execution time: %d ms", Long.valueOf( te - ts ) ) );
144             }
145             catch ( IOException e )
146             {
147                 throw new MojoExecutionException( "?", e );
148             }
149         }
150     }

152     protected abstract void internalExecute()
153         throws MojoExecutionException, MojoFailureException, IOException;

155     protected final BufferedReader createBufferedFileReader( File file, String encoding )
156         throws FileNotFoundException, UnsupportedEncodingException
157     {
158         FileInputStream fis = new FileInputStream( file );
159         InputStreamReader isr = encoding != null ? new InputStreamReader( fis, encoding ) : new InputStreamReader( fis );
160         return new BufferedReader( isr );
161     }
162 
163     protected final BufferedWriter createBufferedFileWriter( File file, String encoding )
164         throws FileNotFoundException, UnsupportedEncodingException
165     {
166         FileOutputStream fos = new FileOutputStream( file );
167         OutputStreamWriter osw = encoding != null ? new OutputStreamWriter( fos, encoding ) : new OutputStreamWriter( fos );
168         return new BufferedWriter( osw );
169     }
170 
171     protected String readFileFirstLine( File file, String encoding )
172         throws IOException
173     {
174         String result = null;
175 
176         BufferedReader is = createBufferedFileReader( file, encoding );
177         try
178         {
179             result = is.readLine();
180         }
181         finally
182         {
183             is.close();
184         }
185         return result;
186     }
187 
188     protected void writeToFile( File file, String encoding, String line )
189         throws IOException
190     {
191         BufferedWriter writer = createBufferedFileWriter( file, encoding );
192         try
193         {
194             writer.write( line );
195         }
196         finally
197         {
198             writer.flush();
199             writer.close();
200         }
201     }

203     protected void createDirectory( File directory, boolean overwrite )
204         throws IOException
205     {
206         if ( directory.exists() )
207         {
208             if ( directory.isDirectory() )
209             {
210                 if ( overwrite )
211                 {
212                     FileUtils.cleanDirectory( directory );
213                 }
214             }
215             else
216             // file if ( directory.isFile() )
217             {
218                 getLog().info( String.format( "Deleting \"%s\" file", directory ) ); // TODO-more descriptive message
219                 if ( !directory.delete() )
220                 {
221                     throw new IOException( String.format( "Cannot delete \"%s\" file", directory.getCanonicalPath() ) );
222                 }
223             }
224         }
225 
226         if ( !directory.exists() )
227         {
228             if ( !directory.mkdirs() )
229             {
230                 throw new IOException( String.format( "Cannot create \"%s\" directory", directory.getCanonicalPath() ) );
231             }
232         }
233     }

235     // Cached classloaders
236     private static final ConcurrentHashMap<String, ClassLoader> cachedClassLoaders = new ConcurrentHashMap<String, ClassLoader>( 2 );
237 
238     private static ClassLoader getCachedClassLoader( String providerId )
239     {
240         return cachedClassLoaders.get( providerId );
241     }
242 
243     private static void setCachedClassLoader( String providerId, ClassLoader classLoader )
244     {
245         cachedClassLoaders.put( providerId, classLoader );
246     }

248     protected Play2Provider getProvider()
249         throws MojoExecutionException
250     {
251         Play2Provider provider = null;
252         if ( !play2Providers.isEmpty() )
253         {
254             provider = getDeclaredProvider();
255         }
256         else
257         {
258             provider = getWellKnownProvider();
259         }
260         return provider;
261     }
262 
263     private Play2Provider getDeclaredProvider()
264         throws MojoExecutionException
265     {
266         if ( play2Providers.size() > 1 )
267         {
268             throw new MojoExecutionException( "Too many Play! providers defined. A maximum of one allowed." );
269         }
270 
271         Map.Entry<String, Play2Provider> providerEntry = play2Providers.entrySet().iterator().next();
272         String providerId = providerEntry.getKey();
273         Play2Provider provider = providerEntry.getValue();
274 
275         getLog().debug( String.format( "Using declared provider \"%s\".", providerId ) );
276 
277         return provider;
278     }

280     private Play2Provider getWellKnownProvider()
281         throws MojoExecutionException
282     {
283         try
284         {
285             String providerId = Play2Providers.getDefaultProviderId( playVersion );
286             ClassLoader providerClassLoader = getCachedClassLoader( providerId );
287             if ( providerClassLoader == null )
288             {
289                 getLog().debug( String.format( "Cached classloader for provider \"%s\" not available.", providerId ) );
290             }
291             else
292             {
293                 if ( providerClassLoader.getParent() == Thread.currentThread().getContextClassLoader() )
294                 {
295                     getLog().debug( String.format( "Using cached classloader for provider \"%s\".", providerId ) );
296                 }
297                 else
298                 {
299                     getLog().debug( String.format( "Invalidated cached classloader for provider \"%s\". Parent classloader changed from %d to %d.",
300                                                    providerId,
301                                                    Integer.valueOf( providerClassLoader.getParent().hashCode() ),
302                                                    Integer.valueOf( Thread.currentThread().getContextClassLoader().hashCode() ) ) );
303                     providerClassLoader = null;
304                 }
305             }
306 
307             if ( providerClassLoader == null )
308             {
309                 Artifact providerArtifact =
310                     getResolvedArtifact( pluginGroupId, "play2-provider-" + providerId, pluginVersion );
311 
312                 Set<Artifact> providerDependencies = getAllDependencies( providerArtifact, null );
313                 List<File> classPathFiles = new ArrayList<File>( providerDependencies.size() + 2 );
314                 classPathFiles.add( providerArtifact.getFile() );
315                 for ( Artifact dependencyArtifact : providerDependencies )
316                 {
317                     classPathFiles.add( dependencyArtifact.getFile() );
318                 }
319 
320                 List<URL> classPathUrls = new ArrayList<URL>( classPathFiles.size() );
321                 for ( File classPathFile : classPathFiles )
322                 {
323                     classPathUrls.add( new URL( classPathFile.toURI().toASCIIString() ) );
324                 }
325 
326                 providerClassLoader =
327                     new URLClassLoader( classPathUrls.toArray( new URL[classPathUrls.size()] ),
328                                         Thread.currentThread().getContextClassLoader() );
329                 getLog().debug( String.format( "Setting cached classloader for provider \"%s\" with parent classloader %d",
330                                                providerId, Integer.valueOf( providerClassLoader.getParent().hashCode() ) ) );
331                 setCachedClassLoader( providerId, providerClassLoader );
332             }
333 
334             ServiceLoader<Play2Provider> providerServiceLoader =
335                 ServiceLoader.load( Play2Provider.class, providerClassLoader );
336             // get first (there should be exactly one)
337             Play2Provider provider = providerServiceLoader.iterator().next();
338 
339             getLog().debug( String.format( "Using autodetected provider \"%s\".", providerId ) );
340 
341             return provider;
342         }
343         catch ( ArtifactNotFoundException e )
344         {
345             throw new MojoExecutionException( "Provider autodetection failed", e );
346         }
347         catch ( ArtifactResolutionException e )
348         {
349             throw new MojoExecutionException( "Provider autodetection failed", e );
350         }
351         catch ( MalformedURLException e )
352         {
353             throw new MojoExecutionException( "Provider autodetection failed", e );
354         }
355     }

359     protected/*private*/ Artifact getResolvedArtifact( String groupId, String artifactId, String version )
360         throws ArtifactNotFoundException, ArtifactResolutionException
361     {
362         Artifact artifact = factory.createArtifact( groupId, artifactId, version, Artifact.SCOPE_RUNTIME, "jar" );
363         resolver.resolve( artifact, remoteRepos, localRepo );
364         return artifact;
365     }
366 
367     protected/*private*/ Set<Artifact> getAllDependencies( Artifact artifact, ArtifactFilter filter )
368         throws ArtifactNotFoundException, ArtifactResolutionException
369     {
370         return getAllDependencies( Collections.singleton( artifact ), filter );
371     }
372 
373     private Set<Artifact> getAllDependencies( Set<Artifact> artifacts, ArtifactFilter filter )
374         throws ArtifactNotFoundException, ArtifactResolutionException
375     {
376         Artifact originatingArtifact = factory.createBuildArtifact( "dummy", "dummy", "1.0", "jar" );
377         ArtifactResolutionResult resolutionResult =
378             resolver.resolveTransitively( artifacts, originatingArtifact, localRepo, remoteRepos, metadataSource,
379                                           filter );
380         return resolutionResult.getArtifacts();
381     }
