39  /**
40   * Base class for Play&#33; distribution packaging mojos.
41   * 
42   * @author <a href="mailto:gslowikowski@gmail.com">Grzegorz Slowikowski</a>
43   */
44  public abstract class AbstractPlay2DistMojo
45      extends AbstractArchivingMojo
46  {
47  
48      /**
49       * Custom config file.
50       * 
51       * @since 1.0.0
52       */
53      @Parameter( property = "config.file" )
54      private File configFile;
55  
56      /**
57       * Distribution additional project artifacts include filter.
58       * 
59       * Comma-separated list of the classifiers of project's additional artifacts
60       * to include.
61       * For example "assets" value means that {@code target/${artifactId}-${version}-assets.jar}
62       * will be added to {@code lib} directory in distribution archive.
63       * 
64       * @since 1.0.0
65       */
66      @Parameter( property = "play2.distClassifierIncludes", defaultValue = "" )
67      private String distClassifierIncludes;
68  
69      /**
70       * Distribution dependency include filter.
71       * 
72       * @since 1.0.0
73       */
74      @Parameter( property = "play2.distDependencyIncludes", defaultValue = "" )
75      private String distDependencyIncludes;
76  
77      /**
78       * Distribution dependency exclude filter.
79       * 
80       * @since 1.0.0
81       */
82      @Parameter( property = "play2.distDependencyExcludes", defaultValue = "" )
83      private String distDependencyExcludes;
84  
85      protected ZipArchiver prepareArchiver()
86          throws IOException, MojoExecutionException, NoSuchArchiverException
87      {
88          ZipArchiver zipArchiver = getZipArchiver();
89  
90          File baseDir = project.getBasedir();
91          File buildDirectory = new File( project.getBuild().getDirectory() );
92  
93          File projectArtifactFile = new File( buildDirectory, project.getBuild().getFinalName() + ".jar" ); // project.getArtifact().getFile();
94          if ( !projectArtifactFile.isFile() )
95          {
96              throw new MojoExecutionException( String.format( "%s not present", projectArtifactFile.getAbsolutePath() ) );
97              // TODO - add info about running "mvn package first"
98          }
99  
100         if ( configFile != null && !configFile.isFile() )
101         {
102             throw new MojoExecutionException( String.format( "%s not present", configFile.getAbsolutePath() ) );
103         }
104 
105         String packageName = project.getArtifactId() + "-" + project.getVersion();
106 
107         String destinationFileName = packageName + "/lib/" + projectArtifactFile.getName();
108         zipArchiver.addFile( projectArtifactFile, destinationFileName );
109 
110         if ( distClassifierIncludes != null && distClassifierIncludes.length() > 0 )
111         {
112             List<String> incl = Arrays.asList( distClassifierIncludes.split( "," ) );
113             for ( String classifier: incl )
114             {
115                 String projectAttachedArtifactFileName =
116                     String.format( "%s-%s.jar", project.getBuild().getFinalName(), classifier.trim() );
117                 File projectAttachedArtifactFile = new File( buildDirectory, projectAttachedArtifactFileName );
118                 if ( !projectAttachedArtifactFile.isFile() )
119                 {
120                     throw new MojoExecutionException( String.format( "%s not present", projectAttachedArtifactFile.getAbsolutePath() ) );
121                 }
122                 destinationFileName = packageName + "/lib/" + projectAttachedArtifactFile.getName();
123                 zipArchiver.addFile( projectAttachedArtifactFile, destinationFileName );
124             }
125         }
126 
127         // preparation
128         Set<?> projectArtifacts = project.getArtifacts();
129 
130         Set<Artifact> excludedArtifacts = new HashSet<Artifact>();
131 
132         AndArtifactFilter dependencyFilter = new AndArtifactFilter();
133         if ( distDependencyIncludes != null && distDependencyIncludes.length() > 0 )
134         {
135             List<String> incl = Arrays.asList( distDependencyIncludes.split( "," ) );
136             PatternIncludesArtifactFilter includeFilter =
137                 new PatternIncludesArtifactFilter( incl, true/* actTransitively */ );
138 
139             dependencyFilter.add( includeFilter );
140         }
141         if ( distDependencyExcludes != null && distDependencyExcludes.length() > 0 )
142         {
143             List<String> excl = Arrays.asList( distDependencyExcludes.split( "," ) );
144             PatternExcludesArtifactFilter excludeFilter =
145                 new PatternExcludesArtifactFilter( excl, true/* actTransitively */ );
146 
147             dependencyFilter.add( excludeFilter );
148         }
149 
150         Set<Artifact> filteredArtifacts = new HashSet<Artifact>(); // TODO-rename to filteredClassPathArtifacts
151         for ( Iterator<?> iter = projectArtifacts.iterator(); iter.hasNext(); )
152         {
153             Artifact artifact = (Artifact) iter.next();
154             if ( artifact.getArtifactHandler().isAddedToClasspath() && !excludedArtifacts.contains( artifact ) )
155             {
156                 // TODO-add checkPotentialReactorProblem( artifact );
157                 if ( dependencyFilter.include( artifact ) )
158                 {
159                     filteredArtifacts.add( artifact );
160                 }
161                 else
162                 {
163                     getLog().debug( artifact.toString() + " excluded" );
164                 }
165             }
166         }
167 
168         // lib
169         for ( Iterator<?> iter = filteredArtifacts.iterator(); iter.hasNext(); )
170         {
171             Artifact artifact = (Artifact) iter.next();
172             File jarFile = artifact.getFile();
173             StringBuilder dfnsb = new StringBuilder();
174             dfnsb.append( artifact.getGroupId() ).append( '.' ).append( artifact.getArtifactId() ).append( '-' ).append( artifact.getVersion() );
175             if ( artifact.getClassifier() != null )
176             {
177                 dfnsb.append( '-' ).append( artifact.getClassifier() );
178             }
179             dfnsb.append( ".jar" ); // TODO-get the real extension?
180             String destFileName = dfnsb.toString();
181             // destinationFileName = ;
182             zipArchiver.addFile( jarFile, packageName + "/lib/" + destFileName/* jarFile.getName() */ );
183         }
184 
185         File linuxStartFile = createLinuxStartFile( buildDirectory );
186         zipArchiver.addFile( linuxStartFile, packageName + "/start" );
187 
188         File windowsStartFile = createWindowsStartFile( buildDirectory );
189         zipArchiver.addFile( windowsStartFile, packageName + "/start.bat" );
190 
191         File readmeFile = new File( baseDir, "README" );
192         if ( readmeFile.isFile() )
193         {
194             zipArchiver.addFile( readmeFile, packageName + "/" + readmeFile.getName() );
195         }
196 
197         if ( configFile != null )
198         {
199             zipArchiver.addFile( configFile, packageName + "/" + configFile.getName() );
200         }
201 
202         checkArchiverForProblems( zipArchiver );
203 
204         return zipArchiver;
205     }
206 
207     private File createLinuxStartFile( File buildDirectory )
208         throws IOException
209     {
210         File result = new File( buildDirectory, "start" );
211         BufferedWriter writer = createBufferedFileWriter( result, "UTF-8" );
212         try
213         {
214             writer.write( "#!/usr/bin/env sh" );
215             writer.newLine();
216             writer.write( "scriptdir=`dirname $0`" );
217             writer.newLine();
218             writer.write( "classpath=$scriptdir/lib/*" );
219             writer.newLine();
220             writer.write( "exec java $* -cp \"$classpath\"" );
221             if ( configFile != null )
222             {
223                 writer.write( " -Dconfig.file=`dirname $0`/" );
224                 writer.write( configFile.getName() );
225             }
226             writer.write( " play.core.server.NettyServer $scriptdir" );
227             writer.newLine();
228         }
229         finally
230         {
231             writer.flush();
232             writer.close();
233         }
234         return result;
235     }
236 
237     private File createWindowsStartFile( File buildDirectory )
238         throws IOException
239     {
240         File result = new File( buildDirectory, "start.bat" );
241         BufferedWriter writer = createBufferedFileWriter( result, "UTF-8" );
242         try
243         {
244             writer.write( "set scriptdir=%~dp0" );
245             writer.newLine();
246             writer.write( "set classpath=%scriptdir%/lib/*" );
247             writer.newLine();
248             writer.write( "java %* -cp \"%classpath%\"" );
249             if ( configFile != null )
250             {
251                 writer.write( " -Dconfig.file=%scriptdir%/" );
252                 writer.write( configFile.getName() );
253             }
254             writer.write( " play.core.server.NettyServer %scriptdir%" );
255             writer.newLine();
256         }
257         finally
258         {
259             writer.flush();
260             writer.close();
261         }
262         return result;
263     }
